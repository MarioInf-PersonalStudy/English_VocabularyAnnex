\newcounter{verbrowcounter}
\newcounter{blocks}

\newcommand{\newIrregularVerb}[8]{%
  \stepcounter{verbrowcounter}%
  \eng{#1}%
  \ifthenelse{\equal{#2}{}}{}{(\emph{US:\usa{#2}})}%
  &
  \eng{#3}%
  \ifthenelse{\equal{#4}{}}{}{(\emph{US:\usa{#4}})}%
  &
  \eng{#5}%
  \ifthenelse{\equal{#6}{}}{}{(\emph{US:\usa{#6}})}%
  &
  \esp{#7}%
  \ifthenelse{\equal{#8}{}}{}{(\emph{Ctx:\esp{#8}})}%
  &
  {\scriptsize \the\numexpr (\theblocks * 20) + \theverbrowcounter}
  \\%
  
  \ifnum\value{verbrowcounter}=20
    \noalign{\vskip 1pt \hrule height 0.8pt \vskip 1pt \hrule height 0.8pt \vskip 1pt}%
    \setcounter{verbrowcounter}{0}%
    \stepcounter{blocks}
  \else
    \cline{1-4}
  \fi
}
