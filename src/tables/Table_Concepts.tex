\newcommand{\newVocabularyConcept}[5]{%
  \stepcounter{rowCounterConcepts}%
  \eng{#1}
  \ifthenelse{\equal{#2}{}}
      {}
      {(\emph{US:\usa{#2}})}
  &
    \textit{#3} & 
    \esp{#4}
    \ifthenelse{\equal{#5}{}}
        {}
        {(\emph{Ctx:\esp{#5}})}
    &
    {\scriptsize \the\numexpr (\thechunkCounterConcepts * 20) + \therowCounterConcepts}
    \\%

    \ifnum\value{rowCounterConcepts}=20
        \noalign{\vskip 1pt \hrule height 0.8pt \vskip 1pt \hrule height 0.8pt \vskip 1pt}%
        \setcounter{rowCounterConcepts}{0}%
        \stepcounter{chunkCounterConcepts}%
    \else
        \cline{1-3}%
    \fi
}
