\newcommand{\newVocabularyConcept}[5]{%
    \eng{#1}
    \ifthenelse{\equal{#2}{}}
        {}
        {(\emph{US:\usa{#2}})}
    &
    \textit{#3} & 
    \esp{#4}
    \ifthenelse{\equal{#5}{}}
        {}
        {(\emph{Ctx:\esp{#5}})}
    \\ 
    \hline
}
